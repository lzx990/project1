    #include<bits/stdc++.h>
#define LL long long
using namespace std;
int n,c,a[214514];
LL ans;
int main()
{
    scanf("%d%d",&n,&c);
    for(int i=1;i<=n;i++)
    {
        scanf("%d",&a[i]);
    }
    sort(a+1,a+1+n);
    int l=1,r=1;
    for(int i=1;i<=n;i++)
    {
        while(a[l]<a[i]-c && l<n) l++;
        while(a[r]<=a[i]-c && r<n) r++;
        if(a[i]-a[l]==c)
        {
            ans+=0LL+r-l;
        }
    }
    printf("%lld\n",ans);
    return 0;
}

    02.Luogu P1638 逛画展¶

    维护指针 \(l,r\) 区间 \(\texttt{[l,r]}\) 即参观的画的画家为 \(a_l,a_{l+1},\dots,a_r\)

    用 \(s\) 表示区间画家个数，\(b\) 数组记录区间内每位画家画的个数

    没有全部画家 \(\texttt{r++}\)，\(\texttt{b[a[r]]++}\)

    如果之前为 \(0\),则 \(\texttt{s++}\)

    有全部画家则 \(\texttt{l++}\)

    之前为 \(1\)，则 \(\texttt{s--}\)

    #include<bits/stdc++.h>
using namespace std;
int n,m,s,b[1919810],a[1919810],ansl,ansr,ans=1919810;
int main()
{
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++)
    {
        scanf("%d",&a[i]);
    }
    int l=1,r=0;
    while(l<=r+1 && r<=n)
    {
        if(s<m)
        {
            r++;
            if(!b[a[r]]) s++;
            b[a[r]]++;
        }
        else
        {
            if(r-l+1<ans)
            {  
                ans=r-l+1;
                ansl=l;
                ansr=r;
            }
            b[a[l]]--;
            if(!b[a[l]]) s--;
            l++;
        }
    }
    printf("%d %d\n",ansl,ansr);
    return 0;
}

    03.Luogu P1115 最大子段和¶

    令 \(f_i\) 表示序列 \(a\) 中，以 \(f_i\) 为右端点的最大子段和

    \(\texttt{f[i]=max(f[i]+a[i],a[i])}\)

    答案为 \(\texttt{max\{f[i]\}}\)

    注意可能有负数

    \(ans\) 需要赋 \(-10^4\) 及以下的值

using namespace std;
int n,a,f,ans=-114514;
int main()
{
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
    {
        scanf("%d",&a);
        f=max(f+a,a);
        ans=max(ans,f);
    }
    printf("%d\n",ans);
    return 0;
}

int n,w,b[610],s;

{
    scanf("%d%d",&n,&w);
    for(int i=1;i<=n;i++)
    {

        b[x]++;
        s=0;

    05.Luogu P26721 求和¶

    Step.1¶

    \(\because y-x=z-y \Rightarrow (x+z)=2y \Rightarrow \frac{x+z}{2}=y\)

    \(\therefore y\) 对答案无影响，只要确保 \(z-x\) 为大于 \(0\) 的偶数即可，即 \(x\) 和 \(z\) 同奇或同偶

    \((x+z) \times (a_x + a_z) = xa_x + xa_z + za_x + za_z\)

    Step.2¶

    不妨设 \(n=5\) 的贡献总和

    i\j 1 2 3 4 5
    1 \(1a_1+1a_2+2a_1+2a_2\) \(1a_1+1a_3+3a_1+3a_3\) \(1a_1+1a_4+4a_1+4a_4\) \(1a_1+1a_5+5a_1+5a_5\)
    2 \(2a_2+2a_3+3a_2+3a_3\) \(2a_2+2a_4+4a_2+4a_4\) \(2a_2+2a_5+5a_2+5a_5\)
    3 \(3a_3+3a_4+4a_3+4a_4\) \(3a_3+3a_5+5a_3+5a_5\)
    4 \(4a_4+4a_5+5a_4+5a_5\)
    5

    可以看出系数是 \(1\) 的项，有 \(4\) 个 \(1a_1\)，还有 \(1(a_2+a_3+a_4+a_5)\)

    可以看出系数是 \(1\) 的项，有 \(4\) 个 \(2a_2\)，还有 \(2(a_1+a_3+a_4+a_5)\)

    \(\dots\)

    可以发现，对于系数为 \(i\) 的项，有 \(n-1\) 个 \(ia_i\) 和 \(i(\sum^5_{j=1}a_j-a_i)\)

    同时要注意自己也算了一次，要扣除

    Step.3¶

    用 \(s1\) 数组记录某颜色编号为奇数或偶数的格子数目

    \(s2\) 某颜色编号奇数或偶数数字之和

    对于每个 \(i\)，贡献为 i*((s2[c[i]][i%2]-a[i])+(a[i]*(s1[c[i]][i%2]-1)))

    #include<bits/stdc++.h>
#define LL long long
using namespace std;
const LL MOD=10007;
int n,m;
LL a[114514],c[114514],s1[114514][2],s2[114514][2],ans;
int main()
{
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++)
    {
        scanf("%lld",&a[i]);
    }
    for(int i=1;i<=n;i++)
    {
        scanf("%lld",&c[i]);
        s1[c[i]][i%2]++;
        s2[c[i]][i%2]=(s2[c[i]][i%2]+a[i])%MOD;
    }
    for(int i=1;i<=n;i++)
    {
        ans=(ans+1LL*i*((s2[c[i]][i%2]-a[i])+(a[i]*(s1[c[i]][i%2]-1)%MOD))%MOD)%MOD;
    }
    printf("%lld\n",(ans+MOD)%MOD);
    return 0;
}

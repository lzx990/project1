    01.Luogu P1177 快速排序¶

    每次分为两个为 \(\frac n 2\) 和 \(\frac {n+1} 2\) 的子序列，分别先进行排序，然后合并

    这样递归直到长度为 \(1\)

    时间复杂度为 \(T(n)=2T(\frac n 2)+O(n)=O(n~log~n)\)

    #include<bits/stdc++.h>
using namespace std;
int b[114514],a[114514],n;
void ms(int a[],int l,int r)
{
    if(l==r) return;
    int mid=(l+r)>>1;
    ms(a,l,mid);
    ms(a,mid+1,r);
    for(int i=l,j=mid+1,k=l;k<=r;k++)
    {
        if(i==mid+1)
        {
            b[k]=a[j++];
        }
        else if(j==r+1)
        {
            b[k]=a[i++];
        }
        else
        {
            b[k]=(a[i]<a[j])?(a[i++]):(a[j++]);
        }
    }
    for(int i=l;i<=r;i++)
    {
        a[i]=b[i];
    }
    return;
}
int main()
{
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
    {
        scanf("%d",&a[i]);
    }
    ms(a,1,n);
    for(int i=1;i<=n;i++)
    {
        printf("%d ",a[i]);
    }
    return 0;
}

    #include<bits/stdc++.h>
#define LL long long
using namespace std;
LL ans;
int b[514514],a[514514],n;
void ms(int a[],int l,int r)
{
    if(l==r) return;
    int mid=(l+r)>>1;
    ms(a,l,mid);
    ms(a,mid+1,r);
    for(int i=l,j=mid+1,k=l;k<=r;k++)
    {
        if(i==mid+1)
        {
            b[k]=a[j++];
        }
        else if(j==r+1)
        {
            b[k]=a[i++];
            ans+=(j-1)-(mid+1)+1LL;
        }
        else if(a[i]<=a[j])
        {
            b[k]=a[i++];
            ans+=(j-1)-(mid+1)+1LL;
        }
        else
        {
            b[k]=a[j++];
        }
    }
    for(int i=l;i<=r;i++)
    {
        a[i]=b[i];
    }
    return;
}
int main()
{
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
    {
        scanf("%d",&a[i]);
    }
    ms(a,1,n);
    printf("%lld\n",ans);
    return 0;
}

    03.Luogu P1966 火柴排队¶

    \((a_i-b_i)^2=a_i^2+b_i^2-2a_ib_i\)

    \(\therefore \sum(a_i-b_i)^2=\sum{a_i^2+b_i^2-2a_ib_i}=\sum{a_i^2}+\sum{b_i^2}-\sum{2a_ib_i}\)

    由于 \(\sum{a_i^2}\) 和 \(\sum{b_i^2}\) 可以确定，所以最大化 \(\sum{2a_ib_i}\) 即可

    第 \(i\) 大的 \(a\) 和第 \(i\) 大的 \(b\) 配在一起是最大的

    第一行交换 \(i\) 与 \(i+1\) 和第二行交换 \(i\) 与 \(i+1\) 是等价的

    所以映射 \(a\) 和 \(b\) 中的每一个数

    然后将 \(b\) 中每一个数映射 \(1,2,3,\dots,n\)

    \(a\) 同样

    就化为每次可以交换相邻两个，问至少多少次排序一个序列

    可以得知每次操作最多减少一个逆序对

    所以归并求有多少个逆序对即可

    #include<bits/stdc++.h>
#define LL long long
using namespace std;
const LL MOD=1e8-3;
LL ans;
int b[114514],a[114514],c[114514],d[114514],li[114514],m,n;
void ms(int a[],int l,int r)
{
    if(l==r) return;
    int mid=(l+r)>>1;
    ms(a,l,mid);
    ms(a,mid+1,r);
    for(int i=l,j=mid+1,k=l;k<=r;k++)
    {
        if(i==mid+1)
        {
            b[k]=a[j++];
        }
        else if(j==r+1)
        {
            b[k]=a[i++];
            ans+=(j-1)-(mid+1)+1LL;
            ans%=MOD;
        }
        else if(a[i]<=a[j])
        {
            b[k]=a[i++];
            ans+=(j-1)-(mid+1)+1LL;
            ans%=MOD;
        }
        else
        {
            b[k]=a[j++];
        }
    }
    for(int i=l;i<=r;i++)
    {
        a[i]=b[i];
    }
    return;
}
int main()
{
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
    {
        scanf("%d",&c[i]);
        li[++m]=c[i];
    }
    sort(li+1,li+1+m);
    m=unique(li+1,li+1+m)-li-1;
    for(int i=1;i<=n;i++)
    {
        int x=lower_bound(li+1,li+1+m,c[i])-li;
        c[i]=x;
    }
    m=0;
    for(int i=1;i<=n;i++)
    {
        scanf("%d",&d[i]);
        li[++m]=d[i];
    }
    sort(li+1,li+1+m);
    m=unique(li+1,li+1+m)-li-1;
    for(int i=1;i<=n;i++)
    {
        int x=lower_bound(li+1,li+1+m,d[i])-li;
        d[i]=x;
    }
    for(int i=1;i<=n;i++)
    {
        a[d[i]]=i;
    }
    for(int i=1;i<=n;i++)
    {
        c[i]=a[c[i]];
    }
    ms(c,1,n);
    printf("%lld\n",ans);
    return 0;
}

    04.Luogu P1226 快速幂¶

    可以将 \(b\) 拆成二进制位

    \(6_{10}=110_2\)

    \(\therefore 3^6=3^4 \times 3^2\)

    由于是二次幂

    \(a^b \times a^b = a^{2b}\)

    所以这样一直乘、当前位为 \(1\) 就累加

    复杂度仅 \(O(log_2b)\)

    #include<bits/stdc++.h>
#define LL long long
using namespace std;
LL a,b,p;
LL kasumi(LL a,LL b,LL p)
{
    LL cnt=a,ans=1%p;
    while(b)
    {
        if(b&1)
        {
            ans=ans*cnt%p;
        }
        cnt=cnt*cnt%p;
        b>>=1;
    }
    return ans;
}
int main()
{
    scanf("%lld%lld%lld",&a,&b,&p);
    printf("%lld^%lld mod %lld=%lld\n",a,b,p,kasumi(a,b,p));
    return 0;
}

    #include<bits/stdc++.h>
using namespace std;
int n,a[214514],ans=INT_MIN;
void ms(int l,int r)
{
    if(l==r)
    {
        ans=max(ans,a[l]);
        return;
    }
    int mid=(l+r)>>1;
    ms(l,mid);
    ms(mid+1,r);
    int ls=INT_MIN,rs=INT_MIN,s=0;
    for(int i=mid;i>=l;i--)
    {
        s+=a[i];
        ls=max(ls,s);
    }
    s=0;
    for(int i=mid+1;i<=r;i++)
    {
        s+=a[i];
        rs=max(rs,s);
    }
    ans=max(ls+rs,ans);
    return;
}
int main()
{
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
    {
        scanf("%d",&a[i]);
    }
    ms(1,n);
    printf("%d\n",ans);
    return 0;
}

    07.Luogu P2880 Balanced Lineup¶

    \(\texttt{st[i][j]}\) 表示 \(a_i,a_{i+1},\dots,a_{i+2^j-2},a_{i+2^j-1}\) 的最大/最小值

    转移方程: \(\texttt{st[i][j]=max/min(st[i][j-1],st[i+} 2^{j-1} \texttt{][j-1])}\)

    查询就要由两个区间拼凑

    令 \(k\) 满足 \(2^k \le r-l+1\) 且最大

    区间 \([l,l+2^k-1]\) 和 \([r-2^k+1,r]\) 就可以拼 \([l,r]\)

    #include<bits/stdc++.h>
using namespace std;
int n,q,lg2[50010],st[2][50010][20];
int M(int op,int a,int b)
{
    if(!op)
    {
        return (a<b)?a:b;
    }
    if(op)
    {
        return (a>b)?a:b;
    }
}
void init()
{
    for(int i=2;i<=n;i++)
    {
        lg2[i]=lg2[(i>>1)]+1;
    }
    for(int k=0;k<=1;k++)
    {
        for(int i=1;(1<<i)<=n;i++)
        {
            for(int j=1;j<=n+1-(1<<i);j++)
            {
                st[k][j][i]=M(k,st[k][j][i-1],st[k][j+(1<<(i-1))][i-1]);
            }
        }
    }
}
int query(int op,int l,int r)
{
    int k=lg2[r-l+1];
    return M(op,st[op][l][k],st[op][r-(1<<k)+1][k]);
}
int main()
{
    scanf("%d%d",&n,&q);
    for(int i=1;i<=n;i++)
    {
        scanf("%d",&st[1][i][0]);
        st[0][i][0]=st[1][i][0];
    }
    init();
    while(q--)
    {
        int l,r;
        scanf("%d%d",&l,&r);
        printf("%d\n",query(1,l,r)-query(0,l,r));
    }
    return 0;
}

    #include<bits/stdc++.h>
using namespace std;
int to[114514][20],s[114514][20],lg2[114514],stD[114514][20],n,q,d[114514],c[114514],r,v;
void init()
{
    for(int i=2;i<=n;i++)
    {
        lg2[i]=lg2[(i>>1)]+1;
    }
    return;
}
int query(int l,int r)
{
    int k=lg2[r-l+1];
    return max(stD[l][k],stD[r-(1<<k)+1][k]);
}
int main()
{
    scanf("%d%d",&n,&q);
    init();
    for(int i=1;i<=n;i++)
    {
        scanf("%d%d",&d[i],&c[i]);
        stD[i][0]=d[i];
    }
    for(int i=1;(1<<i)<=n;i++)
    {
        for(int j=1;j<=n-(1<<i)+1;j++)
        {
            stD[j][i]=max(stD[j][i-1],stD[j+(1<<(i-1))][i-1]);
        }
    }
    c[n+1]=1e9+7;
    for(int i=1;i<n;i++)
    {
        int l=i+1,r=n+1;
        while(l<r)
        {
            int mid=(l+r)>>1;
            if(query(i+1,mid)<=d[i])
            {
                l=mid+1;
            }
            else
            {
                r=mid;
            }
        }
        to[i][0]=l;
        s[i][0]=c[l];
    }
    to[n][0]=n+1;
    s[n][0]=c[n+1];
    for(int i=1;i<=17;i++)
    {
        for(int j=1;j<=n;j++)
        {
            to[j][i]=to[to[j][i-1]][i-1];
            s[j][i]=s[j][i-1]+s[to[j][i-1]][i-1];
        }
    }
    while(q--)
    {
        scanf("%d%d",&r,&v);
        if(v>c[r])
        {
            v-=c[r];
            for(int i=17;i>=0;i--)
            {
                if(v>s[r][i])
                {
                    v-=s[r][i];
                    r=to[r][i];
                }           
            }
            r=to[r][0];
        }
        if(r==n+1) puts("0");
        else printf("%d\n",r);
    }
    return 0;
}
